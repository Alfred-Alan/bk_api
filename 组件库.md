```python
def search_hisotry(request):
    bk_biz_id = request.GET.get("bk_biz_id")
    start_time = request.GET.get("start_time")
    end_time = request.GET.get("end_time")
    current = request.GET.get("current")
    limit = request.GET.get("limit")

    history_obj=History.objects.filter(biz_id=bk_biz_id,create_time__range=(start_time,end_time)).order_by("-create_time")

    paginator = Paginator(history_obj, int(limit))
    data = [history.to_dict() for history in paginator.page(int(current))]

    if len(data)>=1:
        return JsonResponse({"code": 200, "message": "sucess", "data":data,"count":paginator.count})
    else:
        return JsonResponse({"code": 200, "message": "error", "data": []})
```

history model类

```python
class History(models.Model):
    bk_biz_id = models.IntegerField(verbose_name="业务id")
    user_name = models.CharField(max_length=40,verbose_name="用户")
    job_id = models.IntegerField(verbose_name="作业id")
    create_time = models.DateTimeField(verbose_name="创建时间")
    status = models.IntegerField(verbose_name="执行状态")
    status_chooies=[(1,"未执行"),(2,"正在执行"),(3,"执行成功"),(4,"执行失败") ]
    logs = models.TextField(verbose_name="执行日志")

    class Meta:
        db_table="history"

    def to_dict(self,business=[]):
        if business is None:
            business=[{"bk_biz_id":self.bk_biz_id,"bk_biz_name":self.bk_biz_id}][0]
        return {
            "id":self.id,
            "bk_biz_id":[busines["bk_biz_name"]for busines in business if busines["bk_biz_id"]==self.bk_biz_id][0],
            "user_name":self.user_name,
            "sql_name":self.sql_name,
            "job_id":self.job_id,
            "create_time":self.create_time.strftime("%Y-%m-%d %H:%M:%S"),
            "status":[status[1]for status in self.status_chooies if status[0]==self.status][0],
            "logs":eval(str(self.logs)),
        }
```

```python
@task
def check_task_finished(bk_token, bk_biz_id, task_id,ip_detail):
    bk_client.sops.bk_token = bk_token
    status_res = bk_client.sops.get_task_status(bk_biz_id=bk_biz_id, task_id=task_id)

    if status_res.get("result"):
        detail_res = bk_client.sops.get_task_detail(bk_biz_id=bk_biz_id, task_id=task_id)
        for constant in detail_res['outputs']:
            if "JOB任务ID" in constant['name']:
                get_job_detail(bk_token, bk_biz_id, job_instance_id=constant['value'], ip_list=ip_detail[constant['key']])

def get_job_detail(bk_token, bk_biz_id, job_instance_id, ip_list):
    bk_client.job.bk_token = bk_token
    job_status_result = bk_client.job.get_job_instance_status(bk_biz_id, job_instance_id)


    if job_status_result.get("result"):
        step_instance_ids = [{"step_name": step['name'], "step_id": step['step_instance_id']} for step in job_status_result['step']]
        logs = list()
        for ip in ip_list:
            ip_log = {"ip": ip, "step_log": []}
            for step in step_instance_ids:
                job_log_result = bk_client.job.get_job_instance_ip_log(bk_biz_id, job_instance_id, step['step_id'], ip)
                step['log_content'] = job_log_result['data']['log_content']
                ip_log['step_log'].append(step)
            logs.append(ip_log)
        print(logs)
```
